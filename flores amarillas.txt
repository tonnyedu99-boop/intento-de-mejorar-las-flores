<script>
function crearFlor() {
  const flor = document.createElement("div");
  flor.classList.add("flor");
  flor.innerHTML = "🌼";

  flor.style.left = Math.random() * 100 + "vw";
  flor.style.animationDuration = (Math.random() * 3 + 3) + "s";

  document.body.appendChild(flor);

  setTimeout(() => {
    flor.remove();
  }, 6000);
}

setInterval(crearFlor, 300);
